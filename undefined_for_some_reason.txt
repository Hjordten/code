        // Function to create "Save HTML" button for a service section
        function createSaveButton(editorSelector) {
            var button_save = document.createElement("button");
            button_save.textContent = "Save HTML";
            button_save.addEventListener("click", function () {
                var serviceSection = document.querySelector(editorSelector);

                // Get the text content from the textarea-based Rich Text Editor
                var textContent = serviceSection.querySelector(".servicecard-text").value;

                // Preserve the existing image HTML
                var existingImage = serviceSection.querySelector(".servicecard-img img");
                var imageHTML = existingImage ? existingImage.outerHTML : "";

                // Replace the existing content in the servicecard-text div with new text content
                var existingContentText = serviceSection.querySelector(".servicecard-text");
                existingContentText.innerHTML = textContent;

                // Replace the existing content in the servicecard-img div with the preserved image HTML
                var existingContentImg = serviceSection.querySelector(".servicecard-img");
                existingContentImg.innerHTML = imageHTML;
            });

            // Append the button to the body
            document.body.appendChild(button_save);
        }

        // Create "Save HTML" button for the first service section
        createSaveButton(".services-column");
    </script>
